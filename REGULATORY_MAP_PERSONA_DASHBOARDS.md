### **Role Detection Logic:**

```typescript
// Determine which dashboard to show
const getDashboardView = (user: User): DashboardView => {
  const role = user.role; // From user_sites.role
  const userType = user.user_type; // From profiles table
  
  if (role === 'network_admin') return 'entrepreneur';
  if (role === 'site_admin') return 'clinic_owner';
  if (role === 'clinician' && userType === 'therapist') return 'psilocybin_therapist';
  if (role === 'clinician' && userType === 'director') return 'clinical_director';
  if (role === 'clinician') return 'practitioner';
  
  return 'practitioner'; // Default fallback
};
```

### **Dashboard Component Structure:**

```typescript
// Main dashboard router
const RegulatoryMapPage = () => {
  const { user } = useAuth();
  const dashboardView = getDashboardView(user);
  
  switch (dashboardView) {
    case 'psilocybin_therapist':
      return <PsilocybinTherapistDashboard />;
    case 'clinic_owner':
      return <ClinicOwnerDashboard />;
    case 'clinical_director':
      return <ClinicalDirectorDashboard />;
    case 'entrepreneur':
      return <EntrepreneurDashboard />;
    default:
      return <PractitionerDashboard />;
  }
};
```

```sql
CREATE TABLE public.compliance_alerts (
    alert_id UUID PRIMARY KEY,
    user_id UUID REFERENCES auth.users(id),
    alert_type TEXT, -- 'license_expiry', 'form_update', 'regulatory_change'
    severity TEXT CHECK (severity IN ('critical', 'warning', 'info')),
    message TEXT,
    action_url TEXT,
    dismissed BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE public.user_actions_log (
    action_id UUID PRIMARY KEY,
    user_id UUID REFERENCES auth.users(id),
    action_type TEXT, -- 'downloaded_form', 'renewed_license', 'completed_training'
    metadata JSONB,
    created_at TIMESTAMPTZ DEFAULT NOW()
```
